function clone(target) {
  let result = null;
  let cloneQueue = [
    {
      t: target,
      done: (res) => (result = res),

  let map = new WeakMap();

  while (cloneQueue.length > 0) {

          });
        });
        break;
    }

    done(cloneTarget);
  }

  return result;
}
