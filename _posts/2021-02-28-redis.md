# 初识 Redis

    # Macos 下安装Redis
    $ brew install redis

    #安装Brew services list
    $ brew services list

    # 启动Redis
    $ brew services list
    $ brew services start redis  # start restart stop

    #进入redis-cli
    $ redis-cli -a foobared
